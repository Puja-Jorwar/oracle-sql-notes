\# Oracle SQL Selection Assignment



\## WAQTD ALL THE DETAILS OF THE EMPLOYEES

```sql

SELECT \* FROM EMP;

```



\## WAQTD THE ANNUAL SALARY OF THE EMPLOYEE WHOSE NAME IS SMITH

```sql

SELECT SAL \* 12 AS ANNUAL\_SALARY

FROM EMP

WHERE ENAME = 'SMITH';

```



\## WAQTD NAME OF THE EMPLOYEES WORKING AS CLERK

```sql

SELECT ENAME

FROM EMP

WHERE JOB = 'CLERK';

```



\## WAQTD SALARY OF THE EMPLOYEES WHO ARE WORKING AS SALESMAN

```sql

SELECT SAL

FROM EMP

WHERE JOB = 'SALESMAN';

```



\## WAQTD DETAILS OF THE EMP WHO EARNS MORE THAN 2000

```sql

SELECT \*

FROM EMP

WHERE SAL > 2000;

```



\## WAQTD DETAILS OF THE EMP WHOSE NAME IS JONES

```sql

SELECT \*

FROM EMP

WHERE ENAME = 'JONES';

```



\## WAQTD DETAILS OF EMP WHO WAS HIRED AFTER 01-JAN-81

```sql

SELECT \*

FROM EMP

WHERE HIREDATE > '01-JAN-81';

```



\## WAQTD NAME AND SAL ALONG WITH HIS ANNUAL SALARY IF THE ANNUAL SALARY IS MORE THAN 12000

```sql

SELECT ENAME, SAL, SAL \* 12 AS ANNUAL\_SALARY

FROM EMP

WHERE SAL \* 12 > 12000;

```



\## WAQTD EMPNO OF THE EMPLOYEE WHO ARE WORKING IN DEPT 30

```sql

SELECT EMPNO

FROM EMP

WHERE DEPTNO = 30;

```



\## WAQTD ENAME AND HIREDATE IF THEY ARE HIRED BEFORE 1981

```sql

SELECT ENAME, HIREDATE

FROM EMP

WHERE HIREDATE < '01-JAN-81';

```



\## WAQTD DETAILS OF THE EMPLOYEES WORKING AS MANAGER

\## WAQTD DETAILS OF EMPLOYEE HAVING COMMISSION MORE THAN SALARY

\## WAQTD NAME AND SALARY GIVEN TO THE EMPLOYEE IF EMPLOYEE EARNS COMMISSION OF RUPEES 1400

```sql

FROM EMP
```
